CREATE TABLE EMPRESAS (
   ID_EMPRESA INTEGER GENERATED BY DEFAULT AS IDENTITY,
   RAZAO_SOCIAL VARCHAR(100) NOT NULL,
   CNPJ VARCHAR(20),
   INSCRICAO VARCHAR(20),
   ENDERECO VARCHAR(100),
   NUMERO VARCHAR(10),
   BAIRRO VARCHAR(50),
   CIDADE VARCHAR(50),
   ESTADO VARCHAR(2),
   CEP VARCHAR(8),
   TELEFONE VARCHAR(20),
   EMAIL VARCHAR(100),
   IMAGEM_RELATORIO VARCHAR(100));
